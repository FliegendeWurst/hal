516  std::vector<FsmTransition>
517  SolveFsmPlugin::get_state_successors(const z3::expr& prev_state_vec, const z3::expr& next_state_vec, const z3::expr& start_state, const std::map<u32, z3::expr>& external_ids_to_expr)
518  {
519  std::vector<FsmTransition> successor_transitions;
520 
521  z3::solver s = z3::solver(prev_state_vec.ctx());
522 
523  // initialize the previous state vec with the state currently under inspection
524  s.add(prev_state_vec == start_state);
525 
526  // find all possible values for the next state vec
527  while (true)
528  {
529  z3::check_result result = s.check();
530 
531  if (result == z3::sat)
532  {
533  z3::model m = s.get_model();
534  z3::expr n = m.eval(next_state_vec);
535 
536  // check wether the next state only depends on the prev state (is numeral) or contains external inputs
537  if (n.is_numeral())
538  {
539  successor_transitions.push_back({start_state, n, {}});
540  }
541  else
542  {
543  // if the next state bit is dependent on external inputs we find those and brute force all combinations of inputs.
544  // this could be ommitted if we just want to know the transitions without the conditions attached to them.
545  std::vector<u32> relevant_inputs = get_relevant_external_inputs(n, external_ids_to_expr);
546  for (u64 i = 0; i < pow(2, relevant_inputs.size()); i++)
547  {
548  hal::FsmTransition transition = generate_transition_with_inputs(start_state, n, relevant_inputs, i);
549  successor_transitions.push_back(transition);
550  }
551  }
552  s.add(next_state_vec != n);
553  }
554  else
555  {
556  break;
557  }
558  }
559 
560  return successor_transitions;

581  z3::solver s = z3::solver(state.ctx());
582 
583  // generate a mapping from external input net to value and initialize it in the solver.
584  std::map<u32, u8> input_id_to_val;
585  for (u32 i = 0; i < inputs.size(); i++)
586  {
587  u64 val = (input_values >> i) & 0x1;
588  z3::expr val_expr = state.ctx().bv_val(val, 1);
589  z3::expr id_expr = state.ctx().bv_const(std::to_string(inputs.at(i)).c_str(), 1);
590 
591  s.add(id_expr == val_expr);
592  input_id_to_val.insert({inputs.at(i), val});
593  }
594 
595  // evaluate the state reachable under the set input values.
596  s.check();
597  z3::model m = s.get_model();
598  z3::expr eval_state = m.eval(state);
599 
600  return {start_state, eval_state, input_id_to_val};

603  std::vector<FsmTransition> SolveFsmPlugin::merge_transitions(const std::vector<FsmTransition>& transitions)
604  {
605  std::vector<FsmTransition> m_transitions;
606 
607  std::set<u32> already_merged;
608 
609  for (u32 i = 0; i < transitions.size(); i++)
610  {
611  if (already_merged.find(i) != already_merged.end())
612  {
613  continue;
614  }
615 
616  hal::FsmTransition n_transition = transitions.at(i);
617 
618  for (u32 j = 0; j < transitions.size(); j++)
619  {
620  if (i == j)
621  {
622  continue;
623  }
624 
625  if (transitions.at(i).starting_state == transitions.at(j).starting_state && transitions.at(i).end_state == transitions.at(j).end_state)
626  {
627  n_transition = n_transition.merge(transitions.at(j));
628  already_merged.insert(j);
629  }
630  }
631  m_transitions.push_back(n_transition);
632  }
633 
634  log_info("Fsm solver", "Merged transitions. ({} -> {})", transitions.size(), m_transitions.size());
635 
636  return m_transitions;
637  }
