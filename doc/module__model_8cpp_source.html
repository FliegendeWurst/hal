419  Module* ModuleModel::findNetParent(const Net* net){
420  // cannot use Module::get_internal_nets(), because currently that function is implemented so,
421  // that a net can be "internal" to multiple modules at the same depth.
422  // => instead manually search for deepest module, that contains all sources and destinations of net.
423  assert(net);
424  if(net->get_num_of_sources() == 0 && net->get_num_of_destinations() == 0)
425  return nullptr;
426 
427  std::vector<Endpoint*> endpoints = net->get_sources();
428 
429  {
430  std::vector<Endpoint*> destinations = net->get_destinations();
431  endpoints.insert(endpoints.end(), destinations.begin(), destinations.end());
432  }
433 
434  Module* parent = endpoints[0]->get_gate()->get_module();
435  endpoints.erase(endpoints.begin());
436 
437  // might want to split up endpoints, if sources and destinations should be handled differently
438  while(endpoints.size() > 0)
439  {
440  std::vector<Endpoint*>::iterator it = endpoints.begin();
441  while(it != endpoints.end())
442  {
443  if(parent->contains_gate((*it)->get_gate(), true))
444  it = endpoints.erase(it);
445  else
446  ++it;
447  }
448 
449  if(endpoints.size() > 0)
450  parent = parent->get_parent_module();
451  }
452 
453  return parent;
454  }
