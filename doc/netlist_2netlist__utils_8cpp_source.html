612  void remove_buffers(Netlist* netlist, bool analyze_inputs)
613  {
614  u32 num_gates = 0;
615 
616  for (const auto& gate : netlist->get_gates()) {
617  std::vector<Endpoint*> fan_out = gate->get_fan_out_endpoints();
618 
619  GateType* gt = gate->get_type();
620  if (!gt->has_property(GateTypeProperty::combinational) || gt->has_property(GateTypeProperty::power) || gt->has_property(GateTypeProperty::ground)) {
621  // continue if of invalid base type
622  continue;
623  }
624 
625  if (fan_out.size() != 1) {
626  // continue if more than one fan-out net
627  continue;
628  }
629 
630  std::unordered_map<std::string, BooleanFunction> functions = gate->get_boolean_functions();
631  if (functions.size() != 1) {
632  // continue if more than one Boolean function (tri-state?)
633  continue;
634  }
635 
636  Endpoint* out_endpoint = *(fan_out.begin());
637  if (out_endpoint->get_pin() != (functions.begin())->first) {
638  // continue if Boolean function name does not match output pin
639  continue;
640  }
641 
642  std::vector<Endpoint*> fan_in = gate->get_fan_in_endpoints();
643  BooleanFunction func = functions.begin()->second;
644 
645  if (analyze_inputs) {
646  for (Endpoint* ep : fan_in) {
647  auto sources = ep->get_net()->get_sources();
648  if (sources.size() != 1) {
649  break;
650  }
651 
652  if (sources.front()->get_gate()->is_gnd_gate()) {
653  func = func.substitute(ep->get_pin(), BooleanFunction::Value::ZERO);
654  } else if (sources.front()->get_gate()->is_vcc_gate()) {
655  func = func.substitute(ep->get_pin(), BooleanFunction::Value::ONE);
656  }
657  }
658 
659  func = func.optimize_constants();
660  }
661 
662  std::string func_str = func.to_string();
663  std::unordered_set<std::string> in_pins = gt->get_pins_of_direction(PinDirection::input);
664  if (in_pins.find(func_str) != in_pins.end()) {
665  Net* out_net = out_endpoint->get_net();
666 
667  // check all input endpoints and ...
668  for (Endpoint* in_endpoint : fan_in) {
669  Net* in_net = in_endpoint->get_net();
670 
671  if (in_endpoint->get_pin() == func_str) {
672  // reconnect outputs if the input is passed through the buffer
673  for (Endpoint* dst : out_net->get_destinations()) {
674  Gate* dst_gate = dst->get_gate();
675  std::string dst_pin = dst->get_pin();
676  out_net->remove_destination(dst);
677  in_net->add_destination(dst_gate, dst_pin);
678  }
679  } else {
680  // remove the input endpoint otherwise
681  in_net->remove_destination(gate, in_endpoint->get_pin());
682  }
683  }
684 
685  // delete output net and buffer gate
686  netlist->delete_net(out_net);
687  netlist->delete_gate(gate);
688  num_gates++;
689  } else if (func_str == "0" || func_str == "1") {
690  Net* out_net = out_endpoint->get_net();
691 
692  const std::vector<Gate*>& gnd_gates = netlist->get_gnd_gates();
693  const std::vector<Gate*>& vcc_gates = netlist->get_vcc_gates();
694  if (gnd_gates.empty() || vcc_gates.empty()) {
695  continue;
696  }
697  Net* gnd_net = gnd_gates.front()->get_fan_out_nets().front();
698  Net* vcc_net = vcc_gates.front()->get_fan_out_nets().front();
699 
700  for (Endpoint* in_endpoint : fan_in) {
701  // remove the input endpoint otherwise
702  in_endpoint->get_net()->remove_destination(gate, in_endpoint->get_pin());
703  }
704  if (func_str == "0") {
705  for (Endpoint* dst : out_net->get_destinations()) {
706  Gate* dst_gate = dst->get_gate();
707  std::string dst_pin = dst->get_pin();
708  out_net->remove_destination(dst);
709  gnd_net->add_destination(dst_gate, dst_pin);
710  }
711  } else if (func_str == "1") {
712  for (Endpoint* dst : out_net->get_destinations()) {
713  Gate* dst_gate = dst->get_gate();
714  std::string dst_pin = dst->get_pin();
715  out_net->remove_destination(dst);
716  vcc_net->add_destination(dst_gate, dst_pin);
717  }
718  }
719 
720  // delete output net and buffer gate
721  netlist->delete_net(out_net);
722  netlist->delete_gate(gate);
723  num_gates++;
724  }
725  }
726 
727  log_info("netlist_utils", "removed {} buffer gates from the netlist with ID {}.", num_gates, netlist->get_id());
