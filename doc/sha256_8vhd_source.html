132  type SHA_256_HASH_CORE_STATE is ( RESET, IDLE, READ_MSG_BLOCK, PREP_MSG_SCHEDULE_00, PREP_MSG_SCHEDULE_01, PREP_MSG_SCHEDULE_02, PREP_MSG_SCHEDULE_03, HASH_01, HASH_02, HASH_02b, HASH_02c, HASH_03, DONE );

134  signal PREVIOUS_STATE : SHA_256_HASH_CORE_STATE := READ_MSG_BLOCK;

137 
138  --current state logic
139  process(clk, rst)
140  begin
141  if(rst=RESET_VALUE) then
142  CURRENT_STATE <= RESET;
143  elsif(clk'event and clk='1') then
144  CURRENT_STATE <= NEXT_STATE;
145  end if;
146  end process;
147 
148 
149  --next state logic
150  process(CURRENT_STATE, rst, HASH_ROUND_COUNTER, HASH_02_COUNTER, data_ready)
151  begin
152  case CURRENT_STATE is
153  when RESET =>
154  if(rst=RESET_VALUE) then
155  NEXT_STATE <= RESET;
156  else
157  NEXT_STATE <= IDLE;
158  end if;
159  when IDLE =>
160  if(data_ready='1') then
161  NEXT_STATE <= READ_MSG_BLOCK;
162  else
163  NEXT_STATE <= IDLE;
164  end if;
165  when READ_MSG_BLOCK =>
166  NEXT_STATE <= PREP_MSG_SCHEDULE_00;
167  when PREP_MSG_SCHEDULE_00 =>
168  NEXT_STATE <= PREP_MSG_SCHEDULE_01;
169  when PREP_MSG_SCHEDULE_01 =>
170  NEXT_STATE <= PREP_MSG_SCHEDULE_02;
171  when PREP_MSG_SCHEDULE_02 =>
172  NEXT_STATE <= PREP_MSG_SCHEDULE_03;
173  when PREP_MSG_SCHEDULE_03 =>
174  NEXT_STATE <= HASH_01;
175  when HASH_01 =>
176  NEXT_STATE <= HASH_02;
177  when HASH_02 =>
178  if(HASH_02_COUNTER = HASH_02_COUNT_LIMIT) then
179  NEXT_STATE <= HASH_03;
180  else
181  NEXT_STATE <= HASH_02b;
182  end if;
183  when HASH_02b =>
184  NEXT_STATE <= HASH_02c;
185  when HASH_02c =>
186  NEXT_STATE <= HASH_02;
187  when HASH_03 =>
188  if(HASH_ROUND_COUNTER = n_blocks-1) then
189  NEXT_STATE <= DONE;
190  else
191  NEXT_STATE <= IDLE;
192  end if;
193  when DONE =>
194  NEXT_STATE <= DONE; --stay in done state unless reset
195  end case;
196  end process;
197 
198 
199  --hash logic
200  process(clk, rst, CURRENT_STATE)
201  begin
202  if(rst=RESET_VALUE) then
203  HASH_ROUND_COUNTER <= 0;
204  MSG_BLOCK_COUNTER <= 0;
205  elsif(clk'event and clk='1') then
206  a <= a; b <= b; c <= c; d <= d;
207  e <= e; f <= f; g <= g; h <= h;
208  T1 <= T1; T2 <= T2;
209  W <= W; M <= M; HV <= HV;
210  HASH_02_COUNTER <= HASH_02_COUNTER;
211  HASH_ROUND_COUNTER <= HASH_ROUND_COUNTER;
212  case CURRENT_STATE is
213  when RESET =>
214  HV <= HV_INITIAL_VALUES;
215  HASH_02_COUNTER <= 0;
216  HASH_ROUND_COUNTER <= 0;
217  when IDLE => --the IDLE stage is a stall stage, perhaps waiting for new message block to arrive.
218  when READ_MSG_BLOCK =>
219  if(HASH_ROUND_COUNTER = 0) then
220  HV <= HV_INITIAL_VALUES;
221  end if;
222  M <= M_INT;
223  when PREP_MSG_SCHEDULE_00 =>
224  W(0 to 15) <= W_INT(0 to 15);
225  when PREP_MSG_SCHEDULE_01 =>
226  W(16 to 31) <= W_INT(16 to 31);
227  when PREP_MSG_SCHEDULE_02 =>
228  W(32 to 47) <= W_INT(32 to 47);
229  when PREP_MSG_SCHEDULE_03 =>
230  W(48 to 63) <= W_INT(48 to 63);
231  when HASH_01 =>
232  a <= HV(0);
233  b <= HV(1);
234  c <= HV(2);
235  d <= HV(3);
236  e <= HV(4);
237  f <= HV(5);
238  g <= HV(6);
239  h <= HV(7);
240  when HASH_02 =>
241  if(HASH_02_COUNTER = HASH_02_COUNT_LIMIT) then
242  HASH_02_COUNTER <= 0;
243  else
244  --you have to set T1 and T2 in a different state, due to how
245  --VHDL sequential/process statements are evaluated.
246  T1 <= std_logic_vector(unsigned(h) + unsigned(SIGMA_UCASE_1(e)) + unsigned(CH(e, f, g)) + unsigned(K(HASH_02_COUNTER)) + unsigned(W(HASH_02_COUNTER)));
247  T2 <= std_logic_vector(unsigned(SIGMA_UCASE_0(a)) + unsigned(MAJ(a, b, c)));
248  end if;
249  when HASH_02b =>
250  h <= g;
251  g <= f;
252  f <= e;
253  e <= std_logic_vector(unsigned(d) + unsigned(T1));
254  d <= c;
255  c <= b;
256  b <= a;
257  a <= std_logic_vector(unsigned(T1) + unsigned(T2));
258  when HASH_02c =>
259  HASH_02_COUNTER <= HASH_02_COUNTER + 1; --increment counter
260  when HASH_03 =>
261  HV(0) <= std_logic_vector(unsigned(a) + unsigned(HV(0)));
262  HV(1) <= std_logic_vector(unsigned(b) + unsigned(HV(1)));
263  HV(2) <= std_logic_vector(unsigned(c) + unsigned(HV(2)));
264  HV(3) <= std_logic_vector(unsigned(d) + unsigned(HV(3)));
265  HV(4) <= std_logic_vector(unsigned(e) + unsigned(HV(4)));
266  HV(5) <= std_logic_vector(unsigned(f) + unsigned(HV(5)));
267  HV(6) <= std_logic_vector(unsigned(g) + unsigned(HV(6)));
268  HV(7) <= std_logic_vector(unsigned(h) + unsigned(HV(7)));
269  if(HASH_ROUND_COUNTER = n_blocks-1) then
270  HASH_ROUND_COUNTER <= 0;
271  else
272  HASH_ROUND_COUNTER <= HASH_ROUND_COUNTER + 1; --increment counter, read in next message block
273  end if;
274  when DONE =>
275  end case;
276  end if;
277  end process;
