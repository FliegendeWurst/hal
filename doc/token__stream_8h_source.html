444  u32 find_next(const T& match, u32 end = END_OF_STREAM, bool level_aware = true) const
445  {
446  u32 level = 0;
447  for (u32 i = m_pos; i < size() && i < end; ++i)
448  {
449  const auto& Token = at(i);
450  if ((!level_aware || level == 0) && Token == match)
451  {
452  return i;
453  }
454  else if (level_aware && std::find_if(m_increase_level_tokens.begin(), m_increase_level_tokens.end(), [&Token](const auto& x) { return Token == x; }) != m_increase_level_tokens.end())
455  {
456  level++;
457  }
458  else if (level_aware && level > 0
459  && std::find_if(m_decrease_level_tokens.begin(), m_decrease_level_tokens.end(), [&Token](const auto& x) { return Token == x; }) != m_decrease_level_tokens.end())
460  {
461  level--;
462  }
463  }
464  return end;
465  }
