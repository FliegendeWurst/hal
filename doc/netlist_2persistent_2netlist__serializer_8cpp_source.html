437  void serialize(Netlist* nl, rapidjson::Document& document)
438  {
439  rapidjson::Document::AllocatorType& allocator = document.GetAllocator();
440  rapidjson::Value root(rapidjson::kObjectType);
441 
442  root.AddMember("gate_library", nl->get_gate_library()->get_path().string(), allocator);
443  root.AddMember("id", nl->get_id(), allocator);
444  root.AddMember("input_file", nl->get_input_filename().string(), allocator);
445  root.AddMember("design_name", nl->get_design_name(), allocator);
446  root.AddMember("device_name", nl->get_device_name(), allocator);
447 
448  {
449  rapidjson::Value gates(rapidjson::kArrayType);
450  rapidjson::Value global_vccs(rapidjson::kArrayType);
451  rapidjson::Value global_gnds(rapidjson::kArrayType);
452  std::vector<Gate*> sorted = nl->get_gates();
453  std::sort(sorted.begin(), sorted.end(), [](Gate* lhs, Gate* rhs) { return lhs->get_id() < rhs->get_id(); });
454  for (auto gate : sorted)
455  {
456  gates.PushBack(serialize(gate, allocator), allocator);
457  if (nl->is_gnd_gate(gate))
458  {
459  global_gnds.PushBack(gate->get_id(), allocator);
460  }
461  if (nl->is_vcc_gate(gate))
462  {
463  global_vccs.PushBack(gate->get_id(), allocator);
464  }
465  }
466  root.AddMember("gates", gates, allocator);
467  root.AddMember("global_vcc", global_vccs, allocator);
468  root.AddMember("global_gnd", global_gnds, allocator);
469  }
470  {
471  rapidjson::Value nets(rapidjson::kArrayType);
472  rapidjson::Value global_in(rapidjson::kArrayType);
473  rapidjson::Value global_out(rapidjson::kArrayType);
474  std::vector<Net*> sorted = nl->get_nets();
475  std::sort(sorted.begin(), sorted.end(), [](Net* lhs, Net* rhs) { return lhs->get_id() < rhs->get_id(); });
476  for (auto net : sorted)
477  {
478  nets.PushBack(serialize(net, allocator), allocator);
479  if (nl->is_global_input_net(net))
480  {
481  global_in.PushBack(net->get_id(), allocator);
482  }
483  if (nl->is_global_output_net(net))
484  {
485  global_out.PushBack(net->get_id(), allocator);
486  }
487  }
488  root.AddMember("nets", nets, allocator);
489  root.AddMember("global_in", global_in, allocator);
490  root.AddMember("global_out", global_out, allocator);
491  }
492  {
493  rapidjson::Value modules(rapidjson::kArrayType);
494 
495  // module ids are not sorted to preserve hierarchy
496  std::queue<Module*> q;
497  q.push(nl->get_top_module());
498  while (!q.empty())
499  {
500  auto m = q.front();
501  q.pop();
502 
503  modules.PushBack(serialize(m, allocator), allocator);
504 
505  for (auto sm : m->get_submodules())
506  {
507  q.push(sm);
508  }
509  }
510  root.AddMember("modules", modules, allocator);
511  }
512  {
513  rapidjson::Value groupings(rapidjson::kArrayType);
514 
515  std::vector<Grouping*> sorted = nl->get_groupings();
516  std::sort(sorted.begin(), sorted.end(), [](Grouping* lhs, Grouping* rhs) { return lhs->get_id() < rhs->get_id(); });
517  for (auto grouping : sorted)
518  {
519  groupings.PushBack(serialize(grouping, allocator), allocator);
520  }
521  root.AddMember("groupings", groupings, allocator);
522  }
523 
524  document.AddMember("netlist", root, document.GetAllocator());

527  std::unique_ptr<Netlist> deserialize(const rapidjson::Document& document)
528  {
529  if (!document.HasMember("netlist"))
530  {
531  log_critical("netlist_persistent", "file does not include a 'netlist' node");
532  return nullptr;
533  }
534  auto root = document["netlist"].GetObject();
535  assert_availablility("gate_library");
536 
537  auto lib = gate_library_manager::get_gate_library(root["gate_library"].GetString());
538  if (lib == nullptr)
539  {
540  log_critical("netlist_persistent", "error loading gate library '{}'.", root["gate_library"].GetString());
541  return nullptr;
542  }
543 
544  auto nl = std::make_unique<Netlist>(lib);
545 
546  assert_availablility("id");
547  nl->set_id(root["id"].GetUint());
548 
549  assert_availablility("input_file");
550  nl->set_input_filename(root["input_file"].GetString());
551 
552  assert_availablility("design_name");
553  nl->set_design_name(root["design_name"].GetString());
554 
555  assert_availablility("device_name");
556  nl->set_device_name(root["device_name"].GetString());
557 
558  assert_availablility("gates");
559  auto gate_types = nl->get_gate_library()->get_gate_types();
560  for (auto& gate_node : root["gates"].GetArray())
561  {
562  if (!deserialize_gate(nl.get(), gate_node, gate_types))
563  {
564  return nullptr;
565  }
566  }
567 
568  assert_availablility("global_vcc");
569  for (auto& gate_node : root["global_vcc"].GetArray())
570  {
571  nl->mark_vcc_gate(nl->get_gate_by_id(gate_node.GetUint()));
572  }
573 
574  assert_availablility("global_gnd");
575  for (auto& gate_node : root["global_gnd"].GetArray())
576  {
577  nl->mark_gnd_gate(nl->get_gate_by_id(gate_node.GetUint()));
578  }
579 
580  assert_availablility("nets");
581  for (auto& net_node : root["nets"].GetArray())
582  {
583  if (!deserialize_net(nl.get(), net_node))
584  {
585  return nullptr;
586  }
587  }
588 
589  assert_availablility("global_in");
590  for (auto& net_node : root["global_in"].GetArray())
591  {
592  nl->mark_global_input_net(nl->get_net_by_id(net_node.GetUint()));
593  }
594 
595  assert_availablility("global_out");

599  }

615  }

619  for (auto& grouping_node : root["groupings"].GetArray())
