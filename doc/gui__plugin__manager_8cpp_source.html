148

159  {
160  QDir dir(QString::fromStdString(plugDir.string()));
161  if (!dir.exists()) continue;
162  for (QFileInfo info : dir.entryInfoList())
163  {
164  if (!info.isFile()) continue;
165  if (!plugin_manager::has_valid_file_extension(info.absoluteFilePath().toStdString())) continue;
166 
167  QString pluginName = info.baseName();
168  GuiPluginEntry* gpe = pluginEntries.value(pluginName);
169  bool needUpdate = true;
170  if (gpe)
171  {
172  needUpdate = gpe->mFileModified != info.lastModified();
173  }
174  else
175  {
176  gpe = new GuiPluginEntry(info);
177  pluginEntries.insert(pluginName,gpe);
178  }
179 
180  BasePluginInterface* bpif = plugin_manager::get_plugin_instance(pluginName.toStdString(), true, true);
181  if (bpif && !refresh)

183  // plugin already loaded ?
184  bool alreadyLoaded = loadedPlugins.find(pluginName.toStdString())!=loadedPlugins.end();
185  gpe->updateFromLoaded(bpif, alreadyLoaded);
186  log_debug("gui", "GuiPluginManager: '{}' already loaded.", pluginName.toStdString());
187  continue;
188  }
189  else if (gpe->requestLoad())
190  {
191  // user requested this plugin explicitly in previous session
192  bpif = load(pluginName, info.absoluteFilePath());
193  if (!bpif)
194  {
195  gpe->mState = GuiPluginEntry::NotAPlugin; // however, load failed
196  log_warning("gui", "GuiPluginManager: loading of requested plugin '{}' failed.", pluginName.toStdString());
197  }
198  else

200  gpe->updateFromLoaded(bpif, true, info.lastModified());
201  log_info("gui", "GuiPluginManager: '{}' loaded upon user request.", pluginName.toStdString());
202  }
203  continue;
204  }
205 
206  if (!needUpdate)
207  {
208  log_debug("gui", "GuiPluginManager: '{}' not modified, no update required.", pluginName.toStdString());
209  // known plugin, load only when needed
210  if (gpe->isPlugin())
211  gpe->mState = GuiPluginEntry::NotLoaded;
212  continue;
213  }
214 
215  // plugin updated or not known yet, try dummy load to retrieve information
216  bpif = load(pluginName, info.absoluteFilePath());
217  if (!bpif)

219  log_warning("gui", "GuiPluginManager: loading of '{}' failed, is it really a HAL plugin?", pluginName.toStdString());
220  gpe->mState = GuiPluginEntry::NotAPlugin;
221  continue;
222  }
223

227  gpe->mState = GuiPluginEntry::NotAPlugin;
228  }
229  else
230  {
231  log_info("gui", "GuiPluginManager: '{}' loaded to retrieve information about plugin features.", pluginName.toStdString());

233  }
234  }
235  }
236

239  {

241  if (!gpe || gpe->isLoaded()) continue;

246  }
