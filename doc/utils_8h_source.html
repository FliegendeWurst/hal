191  template<typename T>
192  CORE_API std::vector<T> split(const T& s, const char delim, bool obey_brackets = false)
193  {
194  std::vector<T> result;
195  T item = "";
196 
197  if (obey_brackets)
198  {
199  int bracket_level = 0;
200 
201  for (size_t i = 0; i < s.length(); ++i)
202  {
203  char c = s.at(i);
204  switch (c)
205  {
206  case '(':
207  case '{':
208  case '[':
209  ++bracket_level;
210  break;
211  case ')':
212  case '}':
213  case ']':
214  --bracket_level;
215  break;
216  default:
217  break;
218  }
219  if (bracket_level < 0)
220  {
221  bracket_level = 0;
222  }
223  if (c == delim)
224  {
225  // No constant expression, therefore not usable in switch case
226  if (bracket_level == 0)
227  {
228  result.push_back(item);
229  item = "";
230  }
231  else
232  {
233  item.push_back(c);
234  }
235  }
236  else
237  {
238  item.push_back(c);
239  }
240  }
241  if (!item.empty())
242  {
243  result.push_back(item);
244  }
245  }
246  else
247  {
248  std::stringstream ss(s);
249  while (std::getline(ss, item, delim))
250  {
251  result.push_back(item);
252  }
253  }
254  if (s.back() == delim)
255  {
256  result.push_back("");
257  }
258  return result;
259  }

336  template<typename T>
337  CORE_API T replace(const T& str, const T& search, const T& replace)
338  {
339  auto s = str;
340  size_t pos = 0;
341 
342  if (search.empty())
343  {
344  return str;
345  }
346 
347  while ((pos = s.find(search, pos)) != T::npos)
348  {
349  s.replace(pos, search.length(), replace);
350  pos += replace.length();
351  }
352 
353  return s;
