255  std::tuple<QList<TreeItem *>, std::pair<bool, int>, bool> ModulePinsTree::getSelectedPins()
256  {
257  QList<TreeItem*> selectedPins;//ordered
258  QSet<TreeItem*> alreadyProcessedPins;//only for performance purposes
259  bool sameGroup = true;
260  bool onlyPins = true;
261  int groupId = -1;
262  for(auto index : selectionModel()->selectedRows())
263  {
264  TreeItem* item = mPortModel->getItemFromIndex(index);
265  auto itemType = mPortModel->getTypeOfItem(item);
266  if(itemType == ModulePinsTreeModel::itemType::pin)
267  {
268  if(!alreadyProcessedPins.contains(item))
269  {
270  selectedPins.append(item);
271  alreadyProcessedPins.insert(item);
272  }
273  }
274  else if(itemType == ModulePinsTreeModel::itemType::portMultiBit)
275  {
276  onlyPins = false;
277  for(auto pinItem : item->getChildren())
278  {
279  if(!alreadyProcessedPins.contains(pinItem))
280  {
281  selectedPins.append(pinItem);
282  alreadyProcessedPins.insert(pinItem);
283  }
284  }
285  }
286  }
287  //Check if all pins are from the same group (if no pin is selected it is from the same group)
288  if(!selectedPins.isEmpty())
289  {
290  auto mod = gNetlist->get_module_by_id(mModuleID);
291  auto firstPinRes = mod->get_pin_by_id(mPortModel->getIdOfItem(selectedPins.front()));
292  auto firstPin = firstPinRes.get();
293  groupId = firstPin->get_group().first->get_id();
294  for(auto pinTreeItem : selectedPins)
295  {
296  auto pin = mod->get_pin_by_id(mPortModel->getIdOfItem(pinTreeItem)).get();
297  if(groupId != (int)pin->get_group().first->get_id())
298  {
299  sameGroup = false;
300  break;
301  }
302  }
303  groupId = sameGroup ? groupId : -1;
304  }
305  return std::make_tuple(selectedPins, std::make_pair(sameGroup, groupId), onlyPins);
306  }
