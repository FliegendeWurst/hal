26 static u64 bitreverse(u64 n)
27 {
28  static const u64 m0 = 0x5555555555555555LLU;
29  static const u64 m1 = 0x0300c0303030c303LLU;
30  static const u64 m2 = 0x00c0300c03f0003fLLU;
31  static const u64 m3 = 0x00000ffc00003fffLLU;
32  n = ((n >> 1) & m0) | (n & m0) << 1;
33  n = swapbits<u64, m1, 4>(n);
34  n = swapbits<u64, m2, 8>(n);
35  n = swapbits<u64, m3, 20>(n);
36  n = (n >> 34) | (n << 30);
37  return n;

267  }

272  return BooleanFunction();
273  }
274 
275  u64 config = 0;
276  try
277  {
278  config = std::stoull(config_str, nullptr, 16);
279  }
280  catch (std::invalid_argument& ex)
281  {
282  log_error("gate",
283  "LUT gate '{}' with ID {} in netlist with ID {} has invalid configuration string of '{}', which is not a hex value.",
284  m_name,
285  m_id,

287  config_str);
288  return BooleanFunction();
289  }
290

292 
293  if (is_ascending)
294  {
295  config = bitreverse(config) >> (64 - max_config_size);
296  }
297 
298  auto cache_key = std::make_pair(inputs, config);
299  auto& cache = m_internal_manager->m_lut_function_cache;
300 
301  if (auto it = cache.find(cache_key); it != cache.end())
302  {
303  return it->second;
304  }
305 
306  u32 config_size = 0;
307  {
308  u64 tmp = config;
309  while (tmp != 0)
310  {
311  config_size++;
312  tmp >>= 1;
313  }
314  }
315 
316  if (config_size > max_config_size)
317  {
318  log_error("gate",
319  "LUT gate '{}' with ID {} in netlist with ID {} supports a configuration string of up to {} bits, but '{}' comprises {} bits instead.",
320  m_name,
321  m_id,
322  m_internal_manager->m_netlist->get_id(),
323  max_config_size,
324  config_str,
325  config_str.size() * 4);
326  return BooleanFunction();
327  }
328 
329  for (u32 i = 0; config != 0 && i < max_config_size; i++)
330  {
331  u8 bit = (config & 1);
332  config >>= 1;
333  if (bit == 1)
334  {
335  BooleanFunction clause;
336  auto input_values = i;
337  for (auto input : inputs)
338  {
339  if ((input_values & 1) == 1)
340  {
341  clause &= BooleanFunction(input);
342  }
343  else
344  {
345  clause &= ~BooleanFunction(input);
346  }
347  input_values >>= 1;
348  }
349  result |= clause;
350  }
351  }
352 
353  auto f = result.optimize();
354  cache.emplace(cache_key, f);
355  return f;
